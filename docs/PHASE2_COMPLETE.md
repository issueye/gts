- **状态**：✅ 编译通过

### 核心功能

```javascript
let env = require("@std/env");

// 加载 .env 文件
env.load();
env.loadMultiple([".env", ".env.local"]);

// 类型安全访问
let port = env.getInt("PORT", 3000);
let debug = env.getBool("DEBUG", false);
let hosts = env.getArray("API_HOSTS");

// 验证必需变量
env.require(["DATABASE_URL", "API_KEY"]);

// 导出配置
let config = env.toObject({ prefix: "APP_", stripPrefix: true });
```
